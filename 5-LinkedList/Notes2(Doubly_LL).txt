DOUBLY LINKED LIST :
It points towards the next as well as points towards the previous node.
Basic structure of Doubly Linked List:
code:
class Node{
    public:
    int data;
    Node* next;
    Node* back;
    public:
    Node(int data1,Node* next1,Node* back1){
        data = data1;
        next = next1;
        back = back1;
    }
    public:
    Node(int data1,Node* next1,Node* back1){
        data = data1;
        next1 = nullptr;
        back1 = nullptr;
    }
};
